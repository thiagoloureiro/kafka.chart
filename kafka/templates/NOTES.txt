1. Get the application URL by running these commands:
{{- if .Values.kafka.ingress.enabled }}
{{- range $host := .Values.kafka.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.kafka.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.kafka.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "kafka.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.kafka.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "kafka.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "kafka.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.kafka.client_port }}
{{- else if contains "ClusterIP" .Values.kafka.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "kafka.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Kafka brokers are available at:
   Client Port: {{ .Values.kafka.client_port }}
   Internal Port: {{ .Values.kafka.internal_port }}
   
   Connect to Kafka using:
   {{ include "kafka.fullname" . }}:{{ .Values.kafka.client_port }}
   
   Or use individual brokers:
   {{- range $i := until (.Values.kafka.replicas | int) }}
   {{ include "kafka.fullname" $ }}-{{ $i }}.{{ include "kafka.fullname" $ }}-headless.{{ $.Release.Namespace }}.svc.{{ $.Values.clusterDomain }}:{{ $.Values.kafka.client_port }}
   {{- end }}

3. To list Kafka topics:
   kubectl exec -it {{ include "kafka.fullname" . }}-0 -- /opt/kafka/bin/kafka-topics.sh --bootstrap-server localhost:{{ .Values.kafka.client_port }} --list

4. To create a topic:
   kubectl exec -it {{ include "kafka.fullname" . }}-0 -- /opt/kafka/bin/kafka-topics.sh --bootstrap-server localhost:{{ .Values.kafka.client_port }} --create --topic test-topic --partitions 3 --replication-factor {{ .Values.kafka.replicas | int }}
